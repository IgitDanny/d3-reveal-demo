d3.data joins your data into d3

data() lets you do powerful data-driven operations, such as:

  • update() all data elements that are updated
  • enter() all new data elements
  • exit() all old d3 elements

data joins

Data served by your server and ingested by d3 is joined to the most recent data stored by the client.

Visualizing these changes takes the form of 3 types of selections.

Update Selection


// data = an array of data
data = [100, 50, 100, 50, 100, 50, 100, 50];

Enter Selection


// data = an array of data
data = [10, 20];

Exit Selection


// data = an array of data
data = [900, 800, 600, 400, 300, 200, 100];

Apply a change to a property or style of a primitive over time.

Combinations


// g = an SVG 'g' element, h = height in px, w = width in px
g.selectAll('rect')
    .data([0, 10, 20])
  .enter()
    .append('rect')
    .style('fill', 'red') .attr('height', h/4) .attr('width', w/8) .attr('x', -w/2) 
      .attr('y', function(d) { return (10 * d) - h/2;})
    .transition()
      .delay(function(d){ return 100 * (10 + d);})
      .duration(function(d){ return 100 * (10 + d);})
    .attr('x', w*3/8);

Code


// g = an SVG 'g' element, h = height in px, w = width in px
var radius = Math.min(w, h) / 2,
    arc = d3.svg.arc()
      .outerRadius(radius - 10)
      .innerRadius(0);
      
g.append("path")
  .attr("d", arc)
  .style("fill", '#ddd');

g.append("text")
  .attr("transform", function(d) { return "translate(" + arc.centroid(d) + ")"; })
  .attr("dy", ".35em")
  .style("text-anchor", "middle")
  .text(function(d) { return d.data.age; });
